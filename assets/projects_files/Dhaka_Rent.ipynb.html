In [179]:
# Variables
X = df["area"]
Y = df["rent"]
X_bar = np.mean(X)
Y_bar = np.mean(Y)

# Fomula
m = np.sum((X - X_bar)*(Y - Y_bar))/np.sum((X - X_bar)**2)
c = Y_bar - (m * X_bar)

# Result
print("m = ", m)
print("c = ", c)

m =  21.292661473611954
c =  -11469.7354464311
